import pandas as pd
import matplotlib as mpl
import matplotlib.pyplot as plt
from matplotlib.pylab import rcParams
rcParams['figure.figsize'] = 5, 8

import_file1 = 'Sub.1-L.xlsx'
import_file2 = 'Sub.1-R.xlsx'

excel_sheetname01 = 'Sheet1'

df1 = pd.read_excel(import_file1, sheet_name = excel_sheetname01)
df2 = pd.read_excel(import_file2, sheet_name = excel_sheetname01)

df1_Ave=df1['Ave']
df2_Ave=df2['Ave']
plt.plot(df1_Ave, c="dodgerblue")
plt.plot(df2_Ave, c="darkorange")

plt.xlabel("データ数[個]", fontname = "MS Gothic", fontsize = 20)
plt.ylabel("kgf/w", fontname = "MS Gothic", fontsize = 20)
plt.xticks(fontsize = 18)
plt.yticks(fontsize = 18)
plt.xlim(0,60,0)
plt.ylim(0.5,1.0,0.5)
